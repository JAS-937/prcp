print("\n[1/6] Loading precipitation data...")

ds = xr.open_dataset(nc_file)

if 'tp' in ds.variables:
    var_name = 'tp'
elif 'precipitation' in ds.variables:
    var_name = 'precipitation'
else:
    var_name = list(ds.data_vars)[0]

if 'valid_time' in ds.dims:

test_sample = ds[var_name].isel({time_dim: 0}).mean().values

else:
    conversion_factor = 1
    print(f"  Data already in appropriate units")

precip_mm = ds[var_name] * conversion_factor

print("\n[2/6] Loading China boundary...")
shp_file = r'E:\gis\worldmap\world_adm0\world_adm0.shp'

try:
    world = gpd.read_file(shp_file)
    china_candidates = ['China', 'CHINA', 'CHN', 'People']
    china = None
    
    for field in world.columns:
        if world[field].dtype == 'object':
            for candidate in china_candidates:
                mask = world[field].astype(str).str.contains(candidate, case=False, na=False)
                if mask.any():
                    china = world[mask].dissolve()
                    print(f"  ✓ China boundary loaded from field '{field}'")
                    break
        if china is not None:
            break

        raise ValueError("China not found")

    mask = geometry_mask(china_geom, 
                        out_shape=(len(lats), len(lons)),
                        transform=transform,
                        invert=True)

                          coords={'latitude': lats, 'longitude': lons},
                          dims=['latitude', 'longitude'])

for year in unique_years:
    if year < 1981 or year > 2024:
        continue
    
    year_mask = years == year
    year_data = precip_masked.isel({time_dim: year_mask})
    
    if year_mask.sum() < 300:
        continue
    
    rx1day = year_data.max(dim=time_dim)
    year_data_rolled = year_data.rolling({time_dim: 5}, center=False).sum()
    rx5day = year_data_rolled.max(dim=time_dim)
    r20mm = (year_data >= 20).sum(dim=time_dim)
    
    if year <= 2010:
        p95_mask = (years >= 1981) & (years <= year)
        p95_data = precip_masked.isel({time_dim: p95_mask})
        p95_threshold = p95_data.quantile(0.95, dim=time_dim)
    else:
        p95_data = precip_masked.isel({time_dim: baseline_mask})
        p95_threshold = p95_data.quantile(0.95, dim=time_dim)
    
    r95p = year_data.where(year_data > p95_threshold).sum(dim=time_dim)
    
    for i in range(len(lats)):
        for j in range(len(lons)):
            if mask_da.values[i, j]:
                indices_list.append({
                    'year': year,
                    'lat': lats[i],
                    'lon': lons[j],
                    'lat_idx': i,
                    'lon_idx': j,

                    'RX5day': rx5day.values[i, j],
                    'R20mm': r20mm.values[i, j],
                    'R95p': r95p.values[i, j]

        print(f"    Processed year {year}...")

    n_to_sample = min(samples_per_quantile, len(q_indices))
    sampled = np.random.choice(q_indices, size=n_to_sample, replace=False)
    sampled_indices.extend(sampled)

df_sample = df_indices.loc[sampled_indices].copy()
X_sample = df_sample[feature_names].values

print(f"  ✓ Sampled {len(df_sample):,} points (stratified by RX1day)")

K_range = range(2, 6)
inertias = []
db_scores = []

for k in K_range:
    print(f"  Testing K={k}...")
    kmeans = KMeans(n_clusters=k, random_state=42, n_init=20, max_iter=300)
    labels = kmeans.fit_predict(X_sample_scaled)
    inertias.append(kmeans.inertia_)
    db_scores.append(davies_bouldin_score(X_sample_scaled, labels))

fig, axes = plt.subplots(1, 2, figsize=(16, 6))
